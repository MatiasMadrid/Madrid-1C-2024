139static void control_ph(void *pvParameter){
140 uint16_t valorAnalogico_ph = 0;
141 //se declaran los umbrales inferiores y superiores de acuerdo a los valor que entrega el sensor de ph
142 float umbral_inferior = (6.0 * 3.0)/14;
143 float umbral_superior = (6.7 * 3.0)/14;
144 while (1){
145
146 if (sistemaON == true){
147 AnalogInputReadSingle(CH2, &valorAnalogico_ph); // lee el valor analógico que entra por CH2 del sensor de ph
148 //enciende la bomba de ph basico
149 if (valor_ph < umbral_inferior){
150 GPIOOn(bomba_phb.pin);
151 pHCorrecto = false;
152 }
153 //enciende la bomba de ph acido
154 if (valor_ph > umbral_superior){
155 GPIOOn(bomba_pha.pin);
156 pHCorrecto = false;
157 }
158
159 //apaga ambas bombas (estado deseable)
160 if ((valor_ph < umbral_superior) && (valor_ph > umbral_inferior)){
161 GPIOOff(bomba_phb.pin);
162 GPIOOff(bomba_pha.pin);
163 pHCorrecto = true;
164 }
165
166 }
167 valor_ph = (valorAnalogico_ph * 14) / 3;
168 vTaskDelay(PERIODO_LECTURA / portTICK_PERIOD_MS);
169 }
170
171}
