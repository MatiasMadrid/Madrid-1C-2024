123void detectar_ondaR(float p_valor)
124{
125 if ((p_valor > UMBRAL_ONDA_R) && (periodo_RR > muestras_en_periodo_refractario))
126 {
127 FC = 0;
128 FC = 60000000 / (periodo_RR * PERIODO_MUESTREO);
129 periodo_RR = 0;
130 sonarBuzzer = 1;
131 }

153static void procesar_y_enviar_ECG(void *pvParameter)
154{
155 uint16_t valor = 0;
156 char msg[128];
157 char msg_chunk[24];
158 uint8_t i = 0;
159 while (1)
160 {
161 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
162 AnalogInputReadSingle(CH1, &valor);
163 ecg_muestra[i] = valor;
164 valor_actual = valor;
165 periodo_RR++;
166 i++;
167
168 if (i == CHUNK)
169 {
170 HiPassFilter(&ecg_muestra[0], ecg_filt, CHUNK);
171 LowPassFilter(ecg_filt, ecg_filt, CHUNK);
172 strcpy(msg, "");
173
174 for (uint8_t j = 0; j < CHUNK; j++)
175 {
176 detectar_ondaR(ecg_filt[j]);
177 sprintf(msg_chunk, "*G%.2f*", ecg_filt[j]);
178 strcat(msg, msg_chunk);
179 }
180
181 BleSendString(msg);
182 i = 0;
183 }
184 }
185}

222

224 {
225 LedOff(LED_1);

308void app_main(void)
309{
310
311 timer_config_t timer_muestreo = {
312 .timer = TIMER_A,
313 .period = PERIODO_MUESTREO,
314 .func_p = funcTimerMuestreo,
315 .param_p = NULL};
316
317 timer_config_t timer_buzzer = {
318 .timer = TIMER_B,
319 .period = TIME_BUZZER,
320 .func_p = funcTimerBUZZER,
321 .param_p = NULL};
322
323 analog_input_config_t analog_input = {
324 .input = CH1,
325 .mode = ADC_SINGLE,
326 .func_p = NULL,
327 .param_p = NULL};
328
329 ble_config_t ble_configuration = {
330 "ESP_ECG",
331 read_data};
332
333 //Se inicializan los perifericos
334 LedsInit();
335 BuzzerInit(BUZZER);
336 BleInit(&ble_configuration);
337 TimerInit(&timer_muestreo);
338 TimerInit(&timer_buzzer);
339 AnalogInputInit(&analog_input);
340 AnalogOutputInit();
341
342 //Se inician los filtros
343 LowPassInit(SAMPLE_FREQ, 45, ORDER_4);
344 HiPassInit(SAMPLE_FREQ, 1, ORDER_4);
345
346 //Se crean las tareas
347 xTaskCreate(&procesar_y_enviar_ECG, "lee, aplica filtros y envia datos", 4096, NULL, 5, &task_procesar_enviar);
348 xTaskCreate(&sonar, "hace sonar el Buzzer en cada onda R", 4096, NULL, 5, &task_sonar);
349 xTaskCreate(&informar, "informa la frecuancia cardiaca", 4096, NULL, 5, NULL);
350 xTaskCreate(&alarma, "alarma de un estado anormal a travez de los leds", 4096, NULL, 5, NULL);
351
352 //Se arrancan los timers
353 TimerStart(timer_muestreo.timer);
354 TimerStart(timer_buzzer.timer);
355}
