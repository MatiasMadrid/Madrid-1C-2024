145static void procesar_y_enviar_ECG(void *pvParameter)
146{
147 uint16_t valor = 0;
148 char msg[128];
149 char msg_chunk[24];
150 uint8_t i = 0;
151 while (1)
152 {
153 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
154 AnalogInputReadSingle(CH1, &valor);
155 ecg_muestra[i] = valor;
156 valor_actual = valor;
157 periodo_RR++;
158 i++;
159
160 if (i == CHUNK)
161 {
162 HiPassFilter(&ecg_muestra[0], ecg_filt, CHUNK);
163 LowPassFilter(ecg_filt, ecg_filt, CHUNK);
164 strcpy(msg, "");
165
166 for (uint8_t j = 0; j < CHUNK; j++)
167 {
168 detectar_ondaR(ecg_filt[j]);
169 sprintf(msg_chunk, "*G%.2f*", ecg_filt[j]);
170 strcat(msg, msg_chunk);
171 }
172
173 BleSendString(msg);
174 i = 0;
175 }
176 }
