112static void DistanceTask(void *pvParameter){
113 uint16_t distancia;
114 while(1){
115
116 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); /* La tarea espera en este punto hasta recibir una notificación */

120 modificarLed(distancia);
121 if (hold){
122 LcdItsE0803Write(distancia);

124
125 } else if (MedirON == false) {
126 LedsOffAll();
